Ship with the API in minutes

# Install: pip install pulse-sdk from pulse import AgentClient # Initialize with your API key client = AgentClient(api_key="pulse_sk_...") # Run an agent with RAG + tools response = client.run( agent_id="agent_research_v2", messages=[{"role": "user", "content": "Summarize Q3 findings"}], context={"user_id": "usr_42", "team": "product"}, rag={"source": "docs-v2", "top_k": 6}, stream=True ) # Stream chunks + full trace for chunk in response.stream(): print(chunk.text, end="", flush=True) print(response.trace.latency_ms) # 412 print(response.eval.safety_score) # 0.991
